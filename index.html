    <script type="module">

        import { initializeApp } from "https://www.gstatic.com/firebasejs/11.6.1/firebase-app.js";

        import { getFirestore, collection, doc, onSnapshot, addDoc, updateDoc, deleteDoc } from "https://www.gstatic.com/firebasejs/11.6.1/firebase-firestore.js";

        import { getAuth, signInAnonymously, onAuthStateChanged, signInWithCustomToken } from "https://www.gstatic.com/firebasejs/11.6.1/firebase-auth.js";


        const firebaseConfig = JSON.parse(__firebase_config);

        const app = initializeApp(firebaseConfig);

        const auth = getAuth(app);

        const db = getFirestore(app);

        const appId = typeof __app_id !== 'undefined' ? __app_id : 'markus-land-v1';


        let user = null;

        let flaechen = [];


        const initAuth = async () => {

            if (typeof __initial_auth_token !== 'undefined' && __initial_auth_token) {

                await signInWithCustomToken(auth, __initial_auth_token);

            } else {

                await signInAnonymously(auth);

            }

        };


        onAuthStateChanged(auth, (u) => {

            user = u;

            if (user) {

                onSnapshot(collection(db, 'artifacts', appId, 'public', 'data', 'flaechen'), (snapshot) => {

                    flaechen = snapshot.docs.map(doc => ({ id: doc.id, ...doc.data() }));

                    render();

                }, (err) => console.error("Sync error:", err));

            }

        });


        initAuth();


        window.resetFilters = () => {

            document.getElementById('f-loc').value = 'ALL';

            document.getElementById('f-stat').value = 'ALL';

            document.getElementById('f-equi').value = 'ALL';

            document.getElementById('f-prog').value = 'ALL';

            document.getElementById('f-d1').checked = true;

            document.getElementById('f-d2').checked = true;

            render();

        };


        function getLink() {

            let cleanUrl = window.location.href.split('?')[0].split('#')[0];

            if (cleanUrl.startsWith('blob:')) {

                cleanUrl = `https://preview.colab.google/artifacts/${appId}/public/tracker_visualisierung.html`;

            }

            return cleanUrl;

        }


        window.showQR = () => {

            const url = getLink();

            const box = document.getElementById('clean-url-box');

            const qrEl = document.getElementById('qrcode');

            qrEl.innerHTML = '';

            box.value = url;

            new QRCode(qrEl, { text: url, width: 200, height: 200, correctLevel: QRCode.CorrectLevel.H });

            document.getElementById('qr-modal').classList.remove('hidden');

        };


        window.shareWhatsApp = () => {

            const url = getLink();

            const text = encodeURIComponent(`Servus, hier ist der aktuelle Mäh-Plan: ${url}`);

            window.open(`https://wa.me/?text=${text}`, '_blank');

        };


        window.copyCleanLink = () => {

            const el = document.getElementById('clean-url-box');

            el.select();

            document.execCommand('copy');

            const toast = document.getElementById("copy-toast");

            toast.className = "show";

            setTimeout(() => { toast.className = ""; }, 2500);

        };


        window.addFlaeche = async () => {

            const name = document.getElementById('n-name').value;

            if (!name || !user) return;

            await addDoc(collection(db, 'artifacts', appId, 'public', 'data', 'flaechen'), {

                name,

                location: document.getElementById('n-loc').value,

                equipment: document.getElementById('n-equi').value,

                program: document.getElementById('n-prog').value,

                groesse: parseFloat(document.getElementById('n-size').value) || 0,

                order: parseInt(document.getElementById('n-ord').value) || 99,

                day: parseInt(document.getElementById('n-day').value),

                gemaeht: false, geschwadet: false, abgefahren: false,

                g_am: "", s_am: "", a_am: ""

            });

            document.getElementById('n-name').value = '';

            document.getElementById('n-size').value = '';

            document.getElementById('n-ord').value = '';

        };


        window.toggleStatus = async (id, field) => {

            const f = flaechen.find(x => x.id === id);

            const newState = !f[field];

            const timeKey = field === 'gemaeht' ? 'g_am' : (field === 'geschwadet' ? 's_am' : 'a_am');

            const now = new Date();

            const timeStr = `${now.getHours()}:${String(now.getMinutes()).padStart(2,'0')}`;

            

            let updates = { [field]: newState, [timeKey]: newState ? timeStr : "" };

            if (field === 'abgefahren' && newState) {

                if (!f.gemaeht) { updates.gemaeht = true; updates.g_am = timeStr; }

                if (!f.geschwadet) { updates.geschwadet = true; updates.s_am = timeStr; }

            }

            await updateDoc(doc(db, 'artifacts', appId, 'public', 'data', 'flaechen', id), updates);

        };


        window.deleteMe = async (id) => {

            if(confirm('Möchtest du diese Fläche wirklich löschen?')) {

                await deleteDoc(doc(db, 'artifacts', appId, 'public', 'data', 'flaechen', id));

            }

        };


        window.render = () => {

            const container = document.getElementById('job-list');

            const flocs = document.getElementById('f-loc').value;

            const fstat = document.getElementById('f-stat').value;

            const fequi = document.getElementById('f-equi').value;

            const fprog = document.getElementById('f-prog').value;

            const fd1 = document.getElementById('f-d1').checked;

            const fd2 = document.getElementById('f-d2').checked;


            const filtered = flaechen.filter(f => {

                if (flocs !== 'ALL' && f.location !== flocs) return false;

                if (fequi !== 'ALL' && f.equipment !== fequi) return false;

                if (fprog !== 'ALL' && f.program !== fprog) return false;

                if (!((f.day == 1 && fd1) || (f.day == 2 && fd2))) return false;

                if (fstat === 'TODO' && f.gemaeht) return false;

                if (fstat === 'GEMA' && (!f.gemaeht || f.geschwadet)) return false;

                if (fstat === 'GESCHW' && (!f.geschwadet || f.abgefahren)) return false;

                if (fstat === 'ABGEF' && !f.abgefahren) return false;

                return true;

            }).sort((a,b) => a.day - b.day || a.order - b.order);


            container.innerHTML = filtered.map(f => `

                <div class="bg-white p-5 rounded-xl border flex flex-col md:flex-row gap-5 relative shadow-sm card">

                    <div class="absolute top-0 left-0 bg-blue-600 text-white font-bold px-3 py-1 rounded-br-lg text-[10px] uppercase">

                        📁 ${f.location}

                    </div>

                    <button onclick="deleteMe('${f.id}')" class="absolute top-2 right-2 text-gray-300 hover:text-red-500 no-print p-2 transition-colors">

                        <svg class="w-5 h-5" fill="none" stroke="currentColor" viewBox="0 0 24 24"><path stroke-linecap="round" stroke-linejoin="round" stroke-width="2" d="M6 18L18 6M6 6l12 12"></path></svg>

                    </button>

                    <div class="md:w-1/3 pt-4">

                        <div class="flex items-start gap-3">

                            <div class="flex flex-col items-center gap-1">

                                <div class="order-badge">${f.order}</div>

                                <span class="day-badge text-[9px]">T${f.day}</span>

                            </div>

                            <div>

                                <h3 class="text-lg font-bold text-gray-800 leading-tight">${f.name}</h3>

                                <div class="flex gap-1 mt-1">

                                    <span class="px-2 py-0.5 bg-blue-50 text-[10px] rounded text-blue-700 font-bold">${f.program || 'KEIN'}</span>

                                    <span class="px-2 py-0.5 bg-gray-100 text-[10px] rounded text-gray-600 font-bold">${f.groesse} ha</span>

                                </div>

                                <div class="text-[10px] text-orange-600 mt-2 font-medium italic">🚜 ${f.equipment}</div>

                            </div>

                        </div>

                    </div>

                    <div class="flex-1 grid grid-cols-3 gap-3">

                        ${btn(f, 'gemaeht', '✂️', 'Mähen', f.g_am)}

                        ${btn(f, 'geschwadet', '🚜', 'Schwad', f.s_am)}

                        ${btn(f, 'abgefahren', '🚛', 'Fertig', f.a_am)}

                    </div>

                </div>

            `).join('');

            if (filtered.length === 0) container.innerHTML = '<div class="text-center py-20 text-gray-400 italic">Keine Flächen für die aktuelle Auswahl gefunden.</div>';

        };


        function btn(f, field, icon, label, time) {

            const active = f[field];

            return `

                <button onclick="toggleStatus('${f.id}', '${field}')" class="flex flex-col items-center justify-center py-4 rounded-xl border-2 transition-all active:scale-95 ${active ? 'bg-green-50 border-green-500 text-green-700' : 'bg-white border-gray-100 text-gray-400'}">

                    <span class="text-2xl mb-1">${active ? '✅' : icon}</span>

                    <span class="text-[10px] font-bold uppercase tracking-tight">${label}</span>

                    <span class="text-[9px] opacity-60 font-mono mt-1 h-3">${time || ''}</span>

                </button>

            `;

        }

    </script>
